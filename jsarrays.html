Insertar y eliminar elementos

<script>

let p = document.getElementById("inserta");
let insertados = [];
let html_1 = "<strong>Array</strong> ";
let html_2 = "<br>Eliminado:";

function inserta_final(elemento) {
    insertados.push(elemento);
    p.innerHTML = "";
    p.innerHTML = html_1 + insertados.join(", ");
}
function inserta_inicio(elemento) {
    insertados.unshift(elemento);
    p.innerHTML = "";
    p.innerHTML = html_1 + insertados.join(", ");
}
function menos_ultimo() {
    let eliminado = insertados.pop();
    p.innerHTML = "";
    p.innerHTML = html_1 + insertados.join(", ") + html_2 + eliminado;
}
function menos_primero() {
    let eliminado = insertados.shift();
    p.innerHTML = "";
    p.innerHTML = html_1 + insertados.join(", ") + html_2 + eliminado;
}
function reemplaza(elemento) {
    p.innerHTML = "";
    if (insertados.length >= 3) {
        insertados.splice(2, 1, elemento);

        p.innerHTML = html_1 + insertados.join(", ");
    } else {
        p.innerHTML = "No hay suficientes elementos";
    }
}

Ordenar elementos

<script>

let ordena = document.getElementById("ordena");
let ordenados = [];

function inserta(elemento) {
    ordenados.push(elemento);
    ordena.innerHTML = "";
    ordena.innerHTML = html_1 + ordenados.join(", ");
}

function ordenar_array() {
    ordenados.sort();
    ordena.innerHTML = "";
    ordena.innerHTML = html_1 + ordenados.join(", ");
}

function revertir_array() {
    ordenados.reverse();
    ordena.innerHTML = "";
    ordena.innerHTML = html_1 + ordenados.join(", ");
}

function ordenar_array_numeros() {
    ordenados.sort(comparar);
    ordena.innerHTML = "";
    ordena.innerHTML = html_1 + ordenados.join(", ");
}
function comparar(a,b){
    let aa= parseInt(a);
    let bb= parseInt(b);
    return aa-bb;
}
