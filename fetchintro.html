Ejemplo de petición fetch a partir de objeto Request

<script>

const resultado4 = document.getElementById("resultado4");

let cabeceras = new Headers();

let datos = new FormData();
datos.append("user", "usuario");
datos.append("pass", "12345");

console.log("Datos formulario" + datos.toString());
let config = {
    method: "POST",
    headers: cabeceras,
    mode: "cors",
    cache: "default",
    body: datos
}

function servicioPost(url) {
    let miPeticion = new Request(url, config);

    fetch(miPeticion)
            .then((response) => {
                if (response.ok) {
                    response.text()
                            .then((miTexto) => {
                                resultado4.innerHTML = miTexto;
                            });
                } else {
                    console.log('Respuesta de red OK pero error HTTP: ' + response.status);
                }
            })
            .catch(function (error) {
                console.log('Hubo un problema con la petición Fetch: ' + error.message);
            });

}

Se envían datos a un formulario remoto.

Ejemplo de petición fetch a partir de objeto Request y otra con su copia

<script>

function servicioPostCopia(url) {
    let miPeticion = new Request(url, config);
    let miPeticion2 = new Request(url, config);
    let miPeticion3 = miPeticion.clone();

    fetch(miPeticion)
            .then((response) => {
                if (response.ok) {
                    response.text()
                            .then((miTexto) => {
                                resultado5.innerHTML = resultado5.innerHTML + "
Primera petición:" + miTexto;
                            });
                } else {
                    console.log('Respuesta de red OK pero error HTTP: ' + response.status);
                }
            })
            .catch(function (error) {
                console.log('Hubo un problema con la petición Fetch: ' + error.message);
            });

    fetch(miPeticion2)
            .then((response) => {
                if (response.ok) {
                    response.text()
                            .then((miTexto) => {
                                resultado5.innerHTML = resultado5.innerHTML + "
Segunda petición:" + miTexto;
                            });
                } else {
                    console.log('Respuesta de red OK pero error HTTP: ' + response.status);
                }
            })
            .catch(function (error) {
                console.log('Hubo un problema con la petición Fetch: ' + error.message);
            });
    fetch(miPeticion3)
            .then((response) => {
                if (response.ok) {
                    response.text()
                            .then((miTexto) => {
                                resultado5.innerHTML = resultado5.innerHTML + "
Tercera petición:" + miTexto;
                            });
                } else {
                    console.log('Respuesta de red OK pero error HTTP: ' + response.status);
                }
            })
            .catch(function (error) {
                console.log('Hubo un problema con la petición Fetch: ' + error.message);
            });

}
