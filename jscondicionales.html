Condicionales

Ejemplos del condicional if, if/else y if/else/if

<script>

function soloIf(){
    let numero = window.prompt("Dime un número");
    if(isNaN(numero)){
        alert("¡Hey! No es un número");
    }
}

function ifElse(){
    let numero = window.prompt("Dime un número");
    if(isNaN(numero)){
        alert("¡Hey! No es un número");
    }else{
        alert("Buen número el "+numero);
    }
}

function ifElseIf(){
    let numero = window.prompt("Dime un número del 0 al 10");
    if(isNaN(numero)){
        alert("¡Hey! No es un número");
    }else if(numero>=0 && numero<=10){
        alert("Dentro del rango: "+numero);
    }else{
        alert("Fuera del rango");
    }
}



Ejemplos del condicional switch

Una función se puede definir de diversas maneras

<script>

function colores(caso) {
    let lista = document.getElementById("articulo2");
    switch(parseInt(caso)){
        case 1:lista.style.backgroundColor="red";
            break;
        case 2:lista.style.backgroundColor="blue";
            break;
        case 3:lista.style.backgroundColor="green";
            break;
    }
}      

function colores(caso) {
    let lista = document.getElementById("articulo2");
    switch(parseInt(caso)){
        case 1:lista.style.backgroundColor="red";
            break;
        case 2:lista.style.backgroundColor="blue";
            break;
        case 3:lista.style.backgroundColor="green";
            break;
        default:
            lista.style.backgroundColor="lightgrey";
            break;

    }
}
