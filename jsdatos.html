Números

<script>
  let n=10;
  let octal=010;
  let hexadecimal=0x10;
  let flotante=0.1;
  document.write("Número n="+n+"<br>");
  document.write("Número octal="+octal+"<br>");
  document.write("Número hexadecimal="+hexadecimal+"<br>");

  document.write("n/0="+(n/0)+"<br>");
  document.write("-n/0="+(-n/0)+"<br>");
</script>

Operaciones de notación

<script>

let ejemplo=document.getElementById("ejemplo1");

function precision(numero,parametro){
    var opciones = {
        style: "currency",
        currency: "EUR"
    }
    console.log((new Number(numero)).toLocaleString("es-ES"));
    let p=parseInt(parametro);
    let valor=parseFloat(numero);
    let euros = new Number(valor);
    console.log(valor.toLocaleString("es-ES"));
    ejemplo1.innerHTML="toExponential() "+(new Number(valor.toExponential(p))).toLocaleString("es-ES")
        +" toFixed():"+(new Number(valor.toFixed(p))).toLocaleString("es-ES")
        +" toPrecision():"+(new Number(valor.toPrecision(p))).toLocaleString("es-Es")

}
