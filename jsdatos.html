En este ejemplo se muestran ejemplos sencillos del empleo de los tipos de datos de JavaScript

Undefined y null

Abra la consola de JavaScript

<script>
  let t;
  let soynull=null;//soy un objeto null
  console.log("Como no se le ha asigando valor t=" + t);
  t=location.host;
  console.log("Ahora ya si toma valor t=" + t);
  console.log("Ahora una variable null=" + soynull);
  soynull={user:"user1",password:"12345"};
  console.log("Ahora ya no soy null=" + soynull);
</script>

Números

<script>
  let n=10;
  let octal=010;
  let hexadecimal=0x10;
  let flotante=0.1;
  document.write("Número n="+n+"<br>");
  document.write("Número octal="+octal+"<br>");
  document.write("Número hexadecimal="+hexadecimal+"<br>");
  document.write("Número flotante="+flotante+"<br>");
  document.write("n/\'n\'="+(n/'h')+"<br>");
  document.write("n/0="+(n/0)+"<br>");
  document.write("-n/0="+(-n/0)+"<br>");
</script>
